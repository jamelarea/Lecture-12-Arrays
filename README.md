# Lecture-12-Arrays

SLIDE 9, Example initialisation (longhand)

    #include <iostream>
    using namespace std;

    int main()
    {
      int ages[5];

      ages[0] = 19;
      ages[1] = 23;
      ages[2] = 22;
      ages[3] = 30;
      ages[4] = 18;
    }
    
SLIDE 10, Example initialisation (shorthand)

    #include <iostream>
    using namespace std;
    
    int main()
    {
      int array[5] = {19, 23, 22, 30, 18};

	    or

	    int array [] = {19, 23, 22, 30, 18};
    }
